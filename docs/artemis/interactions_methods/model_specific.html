 55    @property
 56    def _compare_ovo(self):
 57        if self.ovo is None:
 58            raise MethodNotFittedException(self.method)
 59        compare_ovo = self.ovo.copy().rename(columns={"root_variable": "Feature 1", "variable": "Feature 2"})
 60        compare_ovo['id'] = compare_ovo[["Feature 1", "Feature 2"]].apply(lambda x: "".join(sorted(x)), axis=1)
 61        return (compare_ovo.groupby("id")
 62                           .agg({"Feature 1": "first", "Feature 2": "first", self.method: "mean"})
 63                           .sort_values("Conditional Smallest Depth Measure",
 64                                        ascending=self.interactions_ascending_order,
